$$\begin{align}
Curl(\mathbf{F})&=\nabla \times \mathbf{F} =(\dfrac{\partial }{\partial x},\dfrac{\partial }{\partial y},\dfrac{\partial }{\partial z} )\times(f_{1},f_{2},f_{3}) = 
\begin{vmatrix}
i & j & k \\
\dfrac{\partial }{\partial x} & \dfrac{\partial }{\partial y} & \dfrac{\partial }{\partial z} \\
f_{1} & f_{2} & f_{3}
\end{vmatrix} 
\end{align}$$
